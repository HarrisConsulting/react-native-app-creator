```bash
#!/bin/bash
# System Prerequisites Checker for React Native Development

check_nodejs() {
    echo "üì¶ Checking Node.js..."
    if ! command -v node &> /dev/null; then
        echo "‚ùå Node.js not found"
        echo "üëâ Install from: https://nodejs.org (LTS version recommended)"
        return 1
    fi
    
    NODE_VERSION=$(node --version | sed 's/v//')
    NODE_MAJOR=$(echo $NODE_VERSION | cut -d. -f1)
    
    if [[ $NODE_MAJOR -lt 18 ]]; then
        echo "‚ö†Ô∏è  Node.js $NODE_VERSION detected (minimum: 18.x)"
        echo "üëâ Update from: https://nodejs.org"
        return 1
    fi
    
    echo "‚úÖ Node.js $NODE_VERSION (compatible)"
    return 0
}

check_xcode() {
    echo "üî® Checking Xcode..."
    if ! command -v xcodebuild &> /dev/null; then
        echo "‚ùå Xcode not found"
        echo "üëâ Install from Mac App Store"
        return 1
    fi
    
    XCODE_VERSION=$(xcodebuild -version | head -1 | awk '{print $2}')
    echo "‚úÖ Xcode $XCODE_VERSION"
    return 0
}
```

```bash
# Enhanced input validation with educational feedback
validate_project_name() {
    local name="$1"
    
    # Check if empty
    if [[ -z "$name" ]]; then
        echo "‚ùå Project name cannot be empty"
        echo "üí° Try something like: MyAwesomeApp, TodoList, or WeatherTracker"
        return 1
    fi
    
    # Check format (letters, numbers, no spaces)
    if [[ ! "$name" =~ ^[A-Za-z][A-Za-z0-9]*$ ]]; then
        echo "‚ùå Invalid format: '$name'"
        echo "‚úÖ Valid format: Must start with letter, then letters/numbers only"
        echo "üí° Examples: MyApp ‚úÖ | my-app ‚ùå | 123App ‚ùå | My App ‚ùå"
        return 1
    fi
    
    # Check length
    if [[ ${#name} -lt 3 ]]; then
        echo "‚ùå Too short: '$name' (minimum 3 characters)"
        return 1
    fi
    
    if [[ ${#name} -gt 30 ]]; then
        echo "‚ùå Too long: '$name' (maximum 30 characters)"
        return 1
    fi
    
    return 0
}

# Enhanced input loop with retry mechanism
get_project_name() {
    local attempts=0
    local max_attempts=5
    
    while [[ $attempts -lt $max_attempts ]]; do
        echo -n "üè∑Ô∏è  Project Name: "
        read PROJECT_NAME
        
        if validate_project_name "$PROJECT_NAME"; then
            echo "‚úÖ Project name: $PROJECT_NAME"
            return 0
        fi
        
        ((attempts++))
        echo "üîÑ Try again (attempt $attempts/$max_attempts)"
        echo ""
    done
    
    echo "‚ùå Maximum attempts reached. Please restart the script."
    exit 1
}
```

```bash
    echo ""

validate_bundle_id() {
    local bundle_id="$1"
    
    # Check format
    if [[ ! "$bundle_id" =~ ^[a-zA-Z0-9.-]+\.[a-zA-Z0-9.-]+\.[a-zA-Z0-9.-]+$ ]]; then
        echo "‚ùå Invalid Bundle ID format"
        echo "‚úÖ Required format: com.company.appname"
        echo "üí° Example: com.yourname.${PROJECT_NAME,,}"
        return 1

    # Check for reserved words
    if [[ "$bundle_id" == *".apple."* ]] || [[ "$bundle_id" == *".com.com."* ]]; then
        echo "‚ùå Bundle ID contains reserved words"
        return 1
    fi
    
    return 0
}
```

```bash
#!/bin/bash
# Interactive Bundle ID Builder for Novice Users

build_bundle_id_interactively() {
    echo "üîß Bundle ID Builder"
    echo "==================="
    
    # Step 1: Domain/Company
    echo "Step 1: Choose your identifier base"
    echo "1) I have a domain (e.g., mycompany.com)"
    echo "2) I want to use my name (e.g., johnsmith)"
    echo "3) I want generic format (com.yourname)"
    
    read -p "Choose option (1-3): " DOMAIN_CHOICE
    
    case $DOMAIN_CHOICE in
        1)
            read -p "Enter your domain (without www): " DOMAIN
            BUNDLE_PREFIX=$(echo "$DOMAIN" | sed 's/\./\n/g' | tac | tr '\n' '.' | sed 's/\.$//')
            ;;
        2)
            read -p "Enter your name (no spaces): " NAME
            BUNDLE_PREFIX="com.${NAME,,}"
            ;;
        3)
            read -p "Enter your identifier: " IDENTIFIER
            BUNDLE_PREFIX="com.${IDENTIFIER,,}"
            ;;
    esac
    
    # Step 2: App identifier
    APP_ID=$(echo "$PROJECT_NAME" | tr '[:upper:]' '[:lower:]')
    SUGGESTED_BUNDLE="$BUNDLE_PREFIX.$APP_ID"
    
    echo ""
    echo "üí° Suggested Bundle ID: $SUGGESTED_BUNDLE"
    read -p "Use this Bundle ID? (Y/n): " USE_SUGGESTED
    
    if [[ "$USE_SUGGESTED" == "n" || "$USE_SUGGESTED" == "N" ]]; then
        read -p "Enter custom Bundle ID: " BUNDLE_ID
    else
        BUNDLE_ID="$SUGGESTED_BUNDLE"
    fi
    
    # Validation
    if validate_bundle_id "$BUNDLE_ID"; then
        echo "‚úÖ Bundle ID: $BUNDLE_ID"
        return 0
    else
        echo "üîÑ Let's try again..."
        build_bundle_id_interactively
    fi
}
```

```bash
# Progress indicator system
show_progress() {
    local current_step="$1"
    local total_steps="$2"
    local description="$3"
    
    local percentage=$((current_step * 100 / total_steps))
    local filled=$((percentage / 5))
    local empty=$((20 - filled))
    
    printf "\r["
    printf "%*s" "$filled" | tr ' ' '‚ñà'
    printf "%*s" "$empty" | tr ' ' '‚ñë'
    printf "] %d%% - %s" "$percentage" "$description"
    
    if [[ $current_step -eq $total_steps ]]; then
        echo ""
        echo "üéâ Complete!"
    fi
}

# Checkpoint system
checkpoint() {
    local checkpoint_name="$1"
    local status="$2"  # "success", "warning", "error"

        "success")

            echo "‚ö†Ô∏è  $checkpoint_name - Warning"
            ;;
        "error")
            echo "‚ùå $checkpoint_name - Failed"

    esac
}
```

```bash
confirm_action() {
    local action="$1"
    local warning="$2"
    
    echo ""
    echo "‚ö†Ô∏è  Confirmation Required"
    echo "Action: $action"
    if [[ -n "$warning" ]]; then
        echo "Warning: $warning"
    fi
    echo ""
    
    while true; do
        read -p "Continue? (y/N): " CONFIRM
        case $CONFIRM in
            [Yy]*)
                return 0
                ;;
            [Nn]*|"")
                echo "‚ùå Action cancelled"
                return 1
                ;;
            *)
                echo "Please answer y or n"
                ;;
        esac
    done
}

# Usage example
if confirm_action "Remove existing project directory" "This will permanently delete all files"; then

fi
```

```bash
create_backup() {
    local file="$1"
    local backup_dir="backups/$(date +%Y%m%d_%H%M%S)"
    
    if [[ -f "$file" ]]; then
        mkdir -p "$backup_dir"
        cp "$file" "$backup_dir/"
        echo "üíæ Backup created: $backup_dir/$(basename "$file")"
    fi
}

# Automatic backup before modifications
backup_and_modify() {
    local file="$1"
    
    if [[ -f "$file" ]]; then
        create_backup "$file"
        echo "üîß Modifying $file..."
    fi
}
```
